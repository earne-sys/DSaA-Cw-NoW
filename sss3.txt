USE employees;

-- 2b) How many currently assigned employees are in each age range
--     and how many employees there are total?

SELECT 'Total' AS age_range, COUNT(DISTINCT e.emp_no) AS employees_count
FROM employees e
INNER JOIN dept_emp de
ON e.emp_no = de.emp_no
WHERE de.to_date = '9999-01-01'

UNION ALL

SELECT '< 60' AS age_range, COUNT(DISTINCT e.emp_no) AS employees_count
FROM employees e
INNER JOIN dept_emp de
ON e.emp_no = de.emp_no
WHERE de.to_date = '9999-01-01'
AND FLOOR(DATEDIFF(CURDATE(), e.birth_date) / 365) < 60

UNION ALL

SELECT '60 - 65' AS age_range, COUNT(DISTINCT e.emp_no) AS employees_count
FROM employees e
INNER JOIN dept_emp de
ON e.emp_no = de.emp_no
WHERE de.to_date = '9999-01-01'
AND FLOOR(DATEDIFF(CURDATE(), e.birth_date) / 365) BETWEEN 60 AND 65

UNION ALL

SELECT '66 - 69' AS age_range, COUNT(DISTINCT e.emp_no) AS employees_count
FROM employees e
INNER JOIN dept_emp de
ON e.emp_no = de.emp_no
WHERE de.to_date = '9999-01-01'
AND FLOOR(DATEDIFF(CURDATE(), e.birth_date) / 365) BETWEEN 66 AND 69

UNION ALL

SELECT '70+' AS age_range, COUNT(DISTINCT e.emp_no) AS employees_count
FROM employees e
INNER JOIN dept_emp de
ON e.emp_no = de.emp_no
WHERE de.to_date = '9999-01-01'
AND FLOOR(DATEDIFF(CURDATE(), e.birth_date) / 365) >= 70;
