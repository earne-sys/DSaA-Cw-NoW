USE employees;

SELECT t.table_name, COUNT(c.column_name) AS columns_count
FROM (
    SELECT 'employees' AS table_name
    UNION
    SELECT 'departments'
    UNION
    SELECT 'dept_emp'
    UNION
    SELECT 'dept_manager'
    UNION
    SELECT 'titles'
    UNION
    SELECT 'salaries'
    UNION
    SELECT 'address'
    UNION
    SELECT 'adr_emp'
) AS t
INNER JOIN information_schema.columns c
ON c.table_schema = 'employees'
AND c.table_name = t.table_name
GROUP BY t.table_name;
