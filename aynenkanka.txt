USE employees;

-- 2a) Simple CTE: employees hired after 2000 with department and title
WITH employees_after_2000 AS
(
    SELECT
        e.emp_no,
        e.first_name,
        e.last_name,
        e.gender,
        e.birth_date,
        e.hire_date,
        d.dept_no,
        d.dept_name,
        t.title
    FROM employees e
    INNER JOIN dept_emp de
        ON e.emp_no = de.emp_no
    INNER JOIN departments d
        ON de.dept_no = d.dept_no
    INNER JOIN titles t
        ON e.emp_no = t.emp_no
    WHERE e.hire_date >= '2000-01-01'
      AND de.to_date = '9999-01-01'
      AND t.to_date = '9999-01-01'
)
SELECT *
FROM employees_after_2000
ORDER BY hire_date, emp_no;
