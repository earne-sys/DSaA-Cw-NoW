-- 3a) Run query BEFORE creating index
-- Check running time and EXPLAIN result

EXPLAIN
SELECT emp_no, first_name, last_name, gender, birth_date, hire_date
FROM employees
WHERE last_name = 'Facello'
  AND first_name = 'Georgi';

SELECT emp_no, first_name, last_name, gender, birth_date, hire_date
FROM employees
WHERE last_name = 'Facello'
  AND first_name = 'Georgi';


-- 3b) Create index for last name + first name

CREATE INDEX idx_emp_last_first
ON employees (last_name, first_name);


-- 3c) Run the same query AFTER creating index
-- It should be faster and EXPLAIN should show idx_emp_last_first

EXPLAIN
SELECT emp_no, first_name, last_name, gender, birth_date, hire_date
FROM employees
WHERE last_name = 'Facello'
  AND first_name = 'Georgi';

SELECT emp_no, first_name, last_name, gender, birth_date, hire_date
FROM employees
WHERE last_name = 'Facello'
  AND first_name = 'Georgi';


-- 3d) Drop index after testing

DROP INDEX idx_emp_last_first ON employees;



USE employees;

-- =====================================================
-- TASK 3
-- Count employees per department whose last name starts with M
-- Check if index improves searching
-- =====================================================


-- 3a) Run BEFORE creating index

EXPLAIN
SELECT d.dept_no, d.dept_name, COUNT(*) AS employee_count
FROM employees e
INNER JOIN dept_emp de ON e.emp_no = de.emp_no
INNER JOIN departments d ON de.dept_no = d.dept_no
WHERE e.last_name LIKE 'M%'
  AND de.to_date = '9999-01-01'
GROUP BY d.dept_no, d.dept_name
ORDER BY d.dept_no;

SELECT d.dept_no, d.dept_name, COUNT(*) AS employee_count
FROM employees e
INNER JOIN dept_emp de ON e.emp_no = de.emp_no
INNER JOIN departments d ON de.dept_no = d.dept_no
WHERE e.last_name LIKE 'M%'
  AND de.to_date = '9999-01-01'
GROUP BY d.dept_no, d.dept_name
ORDER BY d.dept_no;


-- 3b) Create indexes

CREATE INDEX idx_emp_lastname_empno
ON employees (last_name, emp_no);

CREATE INDEX idx_dept_emp_current
ON dept_emp (to_date, emp_no, dept_no);
